CREATE EXTERNAL TABLE cloudtrail_logs (
  Records ARRAY<STRUCT<
    eventVersion: STRING,
    userIdentity: STRUCT<
      type: STRING,
      principalId: STRING,
      arn: STRING,
      accountId: STRING,
      invokedBy: STRING,
      accessKeyId: STRING
    >,
    eventTime: STRING,
    eventSource: STRING,
    eventName: STRING,
    awsRegion: STRING,
    sourceIPAddress: STRING,
    userAgent: STRING,
    errorCode: STRING,
    errorMessage: STRING,
    requestParameters: MAP<STRING, STRING>,
    responseElements: MAP<STRING, STRING>,
    additionalEventData: MAP<STRING, STRING>,
    requestID: STRING,
    eventID: STRING,
    readOnly: BOOLEAN,
    resources: ARRAY<STRUCT<
      ARN: STRING,
      accountId: STRING,
      type: STRING
    >>,
    eventType: STRING,
    apiVersion: STRING,
    managementEvent: STRING,
    recipientAccountId: STRING,
    serviceEventDetails: STRING,
    sharedEventID: STRING,
    vpcEndpointId: STRING
  >>
)
PARTITIONED BY (
  region STRING,
  year STRING,
  month STRING,
  day STRING
)
ROW FORMAT SERDE 'org.openx.data.jsonserde.JsonSerDe'
WITH SERDEPROPERTIES (
  'ignore.malformed.json' = 'true'
)
LOCATION 's3://aws-cloudtrail-logs-304998662528-b7753f74/AWSLogs/'
TBLPROPERTIES (
  'projection.enabled' = 'true',

  'projection.region.type' = 'enum',
  'projection.region.values' = 'us-east-1,us-west-2,sa-east-1',

  'projection.year.type' = 'integer',
  'projection.year.range' = '2020,2035',

  'projection.month.type' = 'integer',
  'projection.month.range' = '1,12',
  'projection.month.digits' = '2',

  'projection.day.type' = 'integer',
  'projection.day.range' = '1,31',
  'projection.day.digits' = '2',

  'storage.location.template' = 's3://aws-cloudtrail-logs-304998662528-b7753f74/AWSLogs/304998662528/CloudTrail/${region}/${year}/${month}/${day}/'
);

Query GetDataAccess:

SELECT 
  r.eventTime,
  r.eventSource,
  r.eventName,
  r.useridentity.type       	AS user_type,
  r.useridentity.principalid	AS principal_id,
  r.useridentity.arn        	AS user_arn,
  r.useridentity.accountid  	AS user_account,
  r.useridentity.invokedby  	AS invoked_by,
  r.sourceipaddress,
  r.useragent,
  r.requestparameters['tablearn']				AS table_arn,
  r.additionaleventdata['requesterservice']		AS requester_service,
  r.additionaleventdata['lakeformationrolesessionname'] AS lakeformation_rolesession_name,
  r.readOnly,
  r.eventType,
  r.managementEvent,
  r.recipientAccountId,
  r.requestID,
  r.eventID
FROM cloudtrail_logs
CROSS JOIN UNNEST(records) AS t(r)
WHERE 
  year = '2026'
  AND month = '07'
  AND day = '16'
  AND r.eventName = 'GetDataAccess'
  AND r.eventSource = 'lakeformation.amazonaws.com'
ORDER BY r.eventTime

Super query

WITH ga AS (
    SELECT
      r.eventTime,
      r.eventSource,
      r.eventName,
      r.useridentity.type        AS user_type,
      r.useridentity.principalid AS principal_id,
      r.useridentity.arn         AS user_arn,
      r.useridentity.accountid   AS user_account,
      r.useridentity.invokedby   AS invoked_by,
      r.sourceipaddress,
      r.useragent,
      r.requestparameters['tablearn'] AS table_arn,
      r.additionaleventdata['requesterservice'] AS requester_service,
      r.additionaleventdata['lakeformationrolesessionname'] AS lakeformation_rolesession_name,
      r.readOnly,
      r.eventType,
      r.managementEvent,
      r.recipientAccountId,
      r.requestid,
      r.eventID
    FROM cloudtrail_logs
    CROSS JOIN UNNEST(records) AS t(r)
    WHERE 
      year = '2026'
      AND month = '07'
      AND day = '17'
      AND r.eventName = 'GetDataAccess'
      AND r.eventSource = 'lakeformation.amazonaws.com'
),
gt AS (
    SELECT 
      r.requestid,
      r.requestparameters['name'] as table_name,
      regexp_extract(
        json_extract_scalar(
            r.requestparameters['auditcontext'],
            '$.additionalauditcontext'
        ),
        'queryId:\s*([^}]+)',
        1
      ) AS query_id,
      r.requestparameters['catalogid'] as catalog_id,
      r.requestparameters['databasename'] as database_name
    FROM cloudtrail_logs
    CROSS JOIN UNNEST(records) AS t(r)
    WHERE 
      year = '2026'
      AND month = '07'
      AND day = '17'
      AND r.eventName = 'GetTable'
      AND r.eventSource = 'glue.amazonaws.com'
)
SELECT 
  ga.*,
  gt.table_name,
  gt.database_name,
  gt.catalog_id,
  gt.query_id
FROM ga
LEFT JOIN gt 
  ON ga.requestid = gt.requestid
ORDER BY ga.eventTime;
